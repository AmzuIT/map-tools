var map = new GMP({
    id: 'mymap',
    lat: 41.3833,
    lng: 2.1833
  }, function (err, instance) {
    if (!err) {
      addMarkers();
    }
  });

  function addMarkers() {
    map.addMarker([{
      lat: 41.3833,
      lng: 2.1833,
      title: 'Barcelona'
    },
      {
        lat: 42.5000,
        lng: 1.5167,
        title: 'Andorra'
      }]);
  }
